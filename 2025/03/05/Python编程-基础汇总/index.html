几大数据结构

  • 列表 List
    有序，可变，可重复
    [1, 2, 2, 3]

  • 元组 Tuple
    有序，不可变，可重复
    (1, 2, 2, 3)

  • 字典 Dict
    无序！可变，键唯一
    {“a”:1, “b”:2}

  • 字符串 Str
    有序，不可变，可重复
    “hello”

  • 集合 Set
    无序，可变，元素唯一(不重复)
    {1,2,3}

可变性决定方法

# 可变对象（Mutable）: 可以修改自身
可变对象 = {
    "列表": ["可以修改元素", "可以添加删除"],
    "字典": ["可以修改值", "可以添加、删除键值对"],
    "集合": ["可以添加删除元素"]
}

# 不可变对象（Immutable）: 不能修改自身
不可变对象 = {
    "字符串": "修改会创建新字符串",
    "元组": "不能修改元素",
    "数字": "不能修改值"
}

字符串

my_str='hello'
# 1. 修改(返回新字符串):
#    - 大小写: lower(), upper(), title(), capitalize()
#    - 替换: replace(old, new)
#    - 去除: strip(), lstrip(), rstrip()
result=my_str.replace('h','H')

# 增加元素
str_new = my_str + '!' #'hello!'

# 删除元素,比如删除最后一个元素
str2 = my_str[:-1]

# 删除空白
my_str.strip() #直接在原字符串上修改但只能删除首尾的空白，字符串中间的空白删不了

#查找: find(), index(), count(), startswith(), endswith()
my_str.count('l') # 2
my_str.index('lo') #3

#判断: isalpha(), isdigit(), isalnum(), isspace() 等; 返回布尔值
my_str.isalpha()  #True
