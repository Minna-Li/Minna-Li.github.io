Pandas基础

数据结构：

Series ：

一维数据（想象Excel中的一列数据，带行索引（有标签））

# 创建一个Series
import pandas as pd
# 默认下标
names1 = pd.Series(['张三', '李四', '王五'])
print(f"names1输出:\n{names1}")

# 自定义下标
names2 = pd.Series(['张三', '李四', '王五'],
                    index = [1,2,3])
print(f"names2输出:\n{names2}") 

print(f"类型：{type(names1)}") #type 是python的内置函数，通过 type() 函数获取变量 names 的数据类型

#调用1号的名字
print(names2[1])
# 输出结果：
names1输出:
0    张三
1    李四
2    王五
dtype: object #print(names)会返回names中元素的数据类型
names2输出:
1    张三
2    李四
3    王五
dtype: object #字符串属于object类
类型：<class 'pandas.core.series.Series'>
张三

type和dtype的区别：

  • type()：返回的是变量本身的类型（它是 pandas Series 对象、列表、字典等），简单来说，type(sth)回答sth是什么容器？（是Series、Dataframe、字典、列表？）
  • .dtype：是 pandas Series/DataFrame 的属性，返回的是 Series 中元素的数据类型（int、float、object 等）；简单来说，sth.dtype回答sth这个容器里装的什么类型的数据

总结： Series的特点就是只有一列value,左边的一列标签是索引(index)

数据的 读 and 写

文件工作流程：
数据产生→SQL数据库存储→用SQL查询分析→导出为csv/json→Excel打开分析→生成json给网页应用

理解：
SQL:一个厨师，负责处理烹饪数据
Excel : 餐厅，优美的呈现数据
json/csv: 数据的包装盒，不同的打包方式，但都是为了在不同地方传递数据，json的主要使用场景是Web前后端数据交换；csv是不同软件之间的导入导出

  • 读
import pandas as pd
# 可以直接读取的三种文件类型
df = pd.read_csv('data.csv')  
df = pd.read_excel('data.xlsx')
df = pd.read_json('data.json')
# 从数据库读取，需要数据库软件，比如mysql
df = pd.read_sql("SELECT * FROM table", conn)
# 完整形式
df = pd.read_sql(
    sql,           # SQL查询语句
    con,           # 数据库连接对象
    index_col=None, # 用哪列作为索引
    coerce_float=True, # 尝试将数字转为浮点数
    params=None,   # 查询参数（防止SQL注入）
    parse_dates=None, # 解析日期列
    chunksize=None  # 分块读取大数据
)
# 应用举例：
import sqlite3  # SQLite数据库
# 步骤1：连接到数据库
conn = sqlite3.connect('example.db')  # 连接到SQLite数据库文件
# 步骤2：用 pandas 执行 SQL 查询
query = "SELECT * FROM users WHERE age > 20"
df = pd.read_sql(query, conn)  # 执行查询，结果转为DataFrame
# 步骤3：关闭连接
conn.close()
  • 写
df.to_csv('data.csv')
df.to_excel('data.xlsx')
df.to_json('data.json')
df.to_sql()

数据筛选和索引

  1. 按列取数据
import pandas as pd
df=pd.read_csv(data.csv)

df['列名'] #取单列
df.列名
df[['列名1','列名2']]
  1. 按行筛选（重点）
# 布尔索引: df[布尔序列]
df[df.age>18]
# 多条件，注意必须加(),否则会报错
df[(df.a>10)&(df.b<5>)]
# 包含sth的内容的筛选
df[df.name.str.contains('sth')]
df_name = df[df['姓名'].str.contains('张')]
print("姓名含张的员工：\n", df_name)
  1. 按位置iloc(index location) / 按标签loc
df.loc['行标签名','列名'] #第一个是必要参数，第二个可选参数
print("loc筛选：员工2的姓名和薪资\n", df.loc['员工2', ['姓名', '薪资']])
print("loc条件筛选：年龄>24的所有列\n", df.loc[df['年龄'] > 24, :])

df.iloc[0,'列名']  #同上，注意按位置必须输入数字而不是'0'
print("iloc切片：前2行，前3列\n", df.iloc[:2, :3])
  1. 去重
    在一个df中（比如一个有人重复多次提交的报名表）常常需要删除重复的行（index）,subset=[‘name’,’age’]这个参数很实用，表示比对列名为’name’,’age’的列，去掉完全相同的index, 参数keep默认保留第一个重复的值.
    注意： NaN视为相同的值
# 基本语法
df.drop_duplicates(
    subset=None,        # 检查哪些列的重复，默认所有列
    keep='first',       # 保留哪个：'first'保留第一个，'last'保留最后一个，False删除所有
    inplace=False,      # 是否修改原DataFrame
    ignore_index=False  # 是否重置索引
)

df.duplicated() #返回布尔序列，重复的为True

# 这两等价
duplicate_mask = df.duplicated()
df_clean1 = df[~duplicate_mask]  # 取反，选择不重复的
df_clean2 = df.drop_duplicates()

缺失值处理

df.dropna(
    axis=0,          # 0或'index'：删除行（默认）；1或'columns'：删除列
    how='any',       # 'any'：有任何缺失就删除；'all'：全部缺失才删除
    thresh=None,     # 保留至少有 thresh 个非缺失值的行/列
    subset=None,     # 只检查指定的列
    inplace=False,   # 是否修改原DataFrame
    ignore_index=False  # 是否重置索引
)
df_drop = df.dropna(subset=['薪资'])  # 删除薪资列有缺失的行
df_fill = df.fillna({'薪资': df['薪资'].mean()})  # 用均值填充薪资缺失值
#注意：没有fillnull()
# 判断空值,返回布尔值的dataframe
df.isna()  #等价于df.isnull
df.notna()

# 自定义函数：对每列计算统计量
def column_stats(col):
    return pd.Series({
        'mean': col.mean(),
        'std': col.std(),
        'min': col.min(),
        'max': col.max()
    })
print(df.apply(column_stats)) #相当于把df的每一列传入column_stats函数

分组与统计（数据分析核心）

分组聚合

# 按城市分组，统计薪资均值
# 1. 按城市分组
df.groupby('城市')
# 单列分组 单类聚合
city_sal = df.groupby('城市')['薪资'].mean()
print("各城市薪资均值：\n", city_sal) #返回Series
# 单类分组 多类聚合
result=df.groupby('城市')[['薪资','列名']].mean() #返回Dataframe
# 多类分组 单类聚合
result = df.groupby(['城市', '部门'])['薪资'].mean()
# 等价写法
result=df.groupby('城市').agg({'薪资'}:'mean')
print(f"\n每个城市-部门组合的平均薪资：{result}") #这是一个多级index的Series

#  多列分组+多指标聚合
group_df = df.groupby(['城市', '高薪标记']).agg({
    '薪资': ['mean', 'sum', 'count'],  # 薪资的均值、总和、数量
    '年龄': 'max'  # 年龄的最大值
})
print("多列分组聚合：\n", group_df)

排序

df.sort_values(
    by,                    # 排序依据的列名（字符串或列表）
    axis=0,                # 0: 按行排序，1: 按列排序（很少用）
    ascending=True,        # True升序，False降序（列表指定多列顺序）
    inplace=False,         # 是否修改原DataFrame
    kind='quicksort',      # 排序算法：'quicksort', 'mergesort', 'heapsort'
    na_position='last',    # 'first'或'last'，NaN值的位置
    ignore_index=False,    # 是否忽略原索引，重建0-based索引
    key=None              # 排序前对值进行转换的函数
)
# 按薪资降序、年龄升序排序
df_sorted = df.sort_values(by=['薪资', '年龄'], ascending=[False, True])
print("先按薪资降序，再按年龄升序：\n", df_sorted)

df.sort_index()  #恢复索引顺序

表的合并和拼接

# merge :类似SQL的join，往右边的拼接，即补充列。需要相同的类名或意义一样的列（比如都表示名字，只不过列名1是姓名，列名2是名字）
result = pd.merge(
    left,           # 左表
    right,          # 右表
    how='inner',    # 连接方式：'inner', 'left', 'right', 'outer', 'cross'
    on=None,        # 连接键（列名）
    left_on=None,   # 左表连接键
    right_on=None,  # 右表连接键
    left_index=False,  # 是否用左表索引
    right_index=False, # 是否用右表索引
    suffixes=('_x', '_y'),  # 重复列后缀
    validate=None   # 验证关系：'one_to_one', 'one_to_many', 'many_to_one', 'many_to_many'
)

#concatenate,numpy中也有，vstack是它的axis=0的特殊情形，pd.concat适合列表之间的简单拼接
result = pd.concat(
    objs,           # 要连接的对象列表
    axis=0,         # 0: 纵向拼接, 1: 横向拼接
    join='outer',   # 'outer'或'inner'
    ignore_index=False,  # 是否忽略原索引
    keys=None,      # 创建多层索引
    sort=False      # 是否排序
)
